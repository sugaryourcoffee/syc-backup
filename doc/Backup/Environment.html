def self.ruby
  file = ".current_environment.tmp"
  system ("env > #{file}")

  lines = []

  File.open(file, 'r') do |f|
    while line = f.gets
      variable = line.chomp.match(%r\A\w+(?=\=)/).to_s
      lines << line.chomp if VARIABLES.find_index variable.upcase
    end
  end

  File.delete file

  lines
end
